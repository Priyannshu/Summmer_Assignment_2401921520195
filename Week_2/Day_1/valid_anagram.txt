class Solution {
public:
    bool isAnagram(string s, string t) {
        unordered_map<char, int> m1;
        unordered_map<char, int> m2;
        for (auto x: s){
            m1[x] +=1;
        }
        for(auto x: t){
            m2[x] +=1;
        }
        if(m1 == m2) return true;
        else return false;
    }
};
